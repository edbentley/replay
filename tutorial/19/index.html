19 - Audio

The only thing our game is missing now is some sound! Let's play the boop sound that replay-starter comes with whenever the bird hits a pipe.

The device parameter has an audio method, which takes a file name and returns a sound object. We can then call play() on it to play a sound.

Lastly we need to ensure the audio file is being loaded by the game in web/index.

108function newPipe(device) {
109 const height = device.size.height + device.size.heightMargin * 2;
110 const randomY = (height - pipeGap * 2) * (device.random() - 0.5);
111
112 return {
113 x: device.size.width + device.size.widthMargin + 50,
114 gapY: randomY,
115 passed: false,
116 };
117}

119function didHitPipe(birdY, size, pipes) {
120 if (
121 birdY - birdHeight / 2 < -(size.height / 2 + size.heightMargin) ||
122 birdY + birdHeight / 2 > size.height / 2 + size.heightMargin
123 ) {
124 // hit bottom or top
125 return true;
126 }
127 for (const pipe of pipes) {
128 if (
129 pipe.x > birdX + birdWidth / 2 + pipeWidth / 2 ||
130 pipe.x < birdX - birdWidth / 2 - pipeWidth / 2
131 ) {
132 // bird isn't at pipe
133 continue;
134 }
135 const {
136 yUpperTop,
137 yUpperBottom,
138 yLowerTop,
139 yLowerBottom,
140 } = getPipeYPositions(size, pipe.gapY);
141 const topPipeRect = {
142 x: pipe.x,
143 y: (yUpperTop + yUpperBottom) / 2,
144 width: pipeWidth,
145 height: yUpperTop - yUpperBottom,
146 };
147 const bottomPipeRect = {
148 x: pipe.x,
149 y: (yLowerTop + yLowerBottom) / 2,
150 width: pipeWidth,
151 height: yLowerTop - yLowerBottom,
152 };
153 // Check a few points at edges of bird
154 const birdPoints = [
155 { x: birdX + birdWidth / 2, y: birdY + birdHeight / 2 },
156 { x: birdX + birdWidth / 2, y: birdY - birdHeight / 2 },
157 { x: birdX, y: birdY + birdHeight / 2 },
158 { x: birdX, y: birdY - birdHeight / 2 },
159 { x: birdX - birdWidth / 2, y: birdY + birdHeight / 2 },
160 { x: birdX - birdWidth / 2, y: birdY - birdHeight / 2 },
161 ];
162 if (
163 birdPoints.some(
164 (point) =>
165 pointInRect(point, topPipeRect) || pointInRect(point, bottomPipeRect)
166 )
167 ) {
168 // Bird hit a pipe!
169 return true;
170 }
171 }
172 return false;
173}

175function pointInRect(point, rect) {
176 return (
177 point.x > rect.x - rect.width / 2 &&
178 point.x < rect.x + rect.width / 2 &&
179 point.y > rect.y - rect.height / 2 &&
180 point.y < rect.y + rect.height / 2
181 );
182}
